C3.js
¶

C3 is a very lightweight wrapper for the Canvas object. It allows you to chain canvas calls, among many other niceties. Its API is heavily-inspired by jQuery.

It has no hard dependencies.

Constructing
¶

The constructor is C3 which creates a C3 object.

C3("#canvas")

You can pass a selector (1), Canvas object (2), a Canvas Context object (3), or a jQuery object (4).

From this, you may chain calls.

 
  function C3(element) {
    return new C3.Object(element);
  }

  C3.Object = function(canvas) {
    if (typeof canvas === 'string') { /* 1 */
      this.canvas = document.querySelector(canvas);
    } if (canvas.getContext) { /* 2 */
      this.canvas = canvas;
    } else if (typeof canvas.save === 'function') { /* 3 */
      this.canvas = canvas.canvas;
    } else if (canvas[0].getContext) { /* 4 */
      this.canvas = canvas[0];
    }

    if (this.canvas) {
      this.cx = this.canvas.getContext('2d');
    }

    this.result = undefined;
    this._style = new C3.Styler();
  };

Canvas method chaining
¶

You can use Canvas functions as chainable calls, so you can chain simple canvas calls.

C3("#canvas")
  .beginPath()
  .moveTo(0, 0)
  .lineTo(10, 10)
  .closePath()
  .fill()

  var methods = ["save", "restore", "scale", "rotate", "translate",
    "transform", "setTransform", "clearRect", "fillRect", "beginPath",
    "closePath", "moveTo", "lineTo", "quadraticCurveTo", "bezierCurveTo",
    "arcTo", "rect", "arc", "fill", "stroke", "clip", "setAlpha",
    "setCompositeOperation", "setLineWidth", "setLineCap", "setLineJoin",
    "setMiterLimit", "clearShadow", "fillText", "strokeText", "setStrokeColor",
    "setFillColor", "strokeRect", "drawImage", "drawImageFromRect",
    "setShadow", "putImageData"];

  for (var i=0; i < methods.length; i++) {
    var method = methods[i];
    (function(method) {
      C3.fn[method] = function() { return this.run(method, arguments); }
    })(method);
  }

fn.attr()
¶

Basic setter/getter for Context properties.

 $("#canvas").attr('globalAlpha', 0.5);
 $("#canvas").attr('globalAlpha');

You can also set multiple attributes at once:

 $("#canvas").attr({
   globalAlpha: 0.5,
   globalCompositeOperation: 'source-over'
 })

You don't often need to use this, as there are aliases available for most attributes already. For instance, those .attr('globalAlpha') calls above are the same as:

 $("#canvas").globalAlpha(0.5);
 $("#canvas").globalAlpha();

  C3.fn.attr = function(key, value) {
    if (typeof key === 'object') {
      for (var attr in key) {
        if (key.hasOwnProperty(attr)) {
          this.setAttr(attr, key[attr]);
        }
      }
    }
    else if (typeof value === 'undefined') {
      return this.getAttr(key);
    } else {
      this.setAttr(key, value);
      return this;
    }
  };
¶

Internally, all attribute get calls pass through here.

 
  C3.fn.getAttr = function(key) {
    if (this.cx) {
      return this.cx[key];
    }
  };
¶

Internally, all attribute set calls pass through here.

 
  C3.fn.setAttr = function(key, value) {
    if (this.cx) {
      this.cx[key] = value;
    }
    return this;
  };

fn.run()
¶

Performs a function call method to the Canvas Context, passing the arguments array args.

 C3("#canvas").run("stroke")

 // Same as canvas.getContext("2d").stroke();

You can pass arguments as an array.

C3("#canvas").run("moveTo", [10, 40]);

// Same as canvas.getContext("2d").moveTo(10, 40);

You can get the result via .result.

 C3("#canvas").run("getImageData", [0,1,2,3]).result;

You don't often need to use this, as there are aliases available for most attributes already. For instance, the .run("moveTo", [10, 40]) call above is the same as:

 $("#canvas").moveTo(10, 40);

Internally, all canvas context calls pass through here.

 
  C3.fn.run = function(method, args) {
    if (this.cx) {
      this.result = this.cx[method].apply(this.cx, args);
    } else {
      this.result = undefined;
    }

    return this;
  };

How to draw
¶

You will usually need to use .style(), some drawing functions, and .paint() in that order:

The .style() sets the ink styles. The drawing functions (like .rect(), .line(), ...) will draw paths, while the .paint() call will paint by filling/stroking the styles set earlier.

 
  /* Example:

     C3("#canvas")
       .style({ fillColor: 'black' })
       .rect(x,y,w,h)
       .paint()

   */
¶

Styles
¶

Provisions for styling are provided via .paint(), .style() and .clearStyle().
¶

fn.style()
¶

Sets style. Be sure to invoke paint() after drawing.

C3("#canvas")
  .style({ fillStyle: 'red' })
  .rect(...)
  .paint()

When invoked without arguments, returns the styles previously set by .style().

$("#canvas").style();
//=> {fillStyle:'red'}

  C3.fn.style = function(styles) {
    if (styles) {
      this._style.add(this, styles);
      return this;
    } else {
      return this._style.styles;
    }
  };
¶

fn.clearStyle()
¶

Resets the styles set using .style().

C3("#canvas")
  .style({ strokeStyle: 'blue', fillStyle: 'red' })
  .line([[0,10], [100,10], [200,200]])
  .paint()

  .clearStyle()
  .style({ strokeStyle: 'green' })
  .line([[0,10], [100,10]])
  .paint()

  C3.fn.clearStyle = function() {
    this._style.reset(this);
    return this;
  };
¶

fn.paint()
¶

Applies the styles by performing fill() and stroke() as needed.

C3("#canvas")
  .style({ fillStyle: 'red' })
  .rect(0, 0, 20, 20)
  .paint()

This example makes paint() invoke fill(), which applies the styles.

 
  C3.fn.paint = function() {
    this._style.paint(this);
    return this;
  };

Drawing functions
¶

Aside from the basic canvas drawing functions, C3 provides these helpful helpers.
¶

fn.clear()
¶

Clears the canvas area on given x, y, w, h rectangle. When no arguments are given, clears the entire canvas.

C3("#canvas").clear()
C3("#canvas").clear(0, 0, 30, 30)

  C3.fn.clear = function(x, y, w, h) {
    if (x != null) {
      this.cx.clearRect(0, 0, this.width(), this.height());
    } else {
      this.cx.clearRect(x, y, w, h);
    }
    return this;
  };
¶

fn.line()
¶

Draws straight lines.

C3("#canvas")
  .line([[0,0], [10,10]])
  .line([[0,0], [10,10], [20,20], ...])

  C3.fn.line = function(list) {
    var self = this;
    this.beginPath();

    for (var index in list) {
      if (list.hasOwnProperty(index)) {
        var xy = list[index];
        var method = (index == 0 ? 'moveTo' : 'lineTo');
        self.run(method, xy);
      }
    }

    this.closePath();
    return this;
  };

Command compilation
¶

You can compile Canvas calls into an anonymous function using .compile() and .compileEnd(). Useful for 'caching' things, or maybe using c3 in a web worker.

var fn = C3("#canvas")
  .compile()
  .style({ fillStyle: 'red' })
  .line([[0,0], [1,10])
  .paint()
  .compileEnd()
¶

The resulting function takes a Canvas element.

fn(canvas);
¶

The result is a flat, anonymous function. It will have no closure dependencies, so it's perfectly useable as a string, as well.

console.log(fn.toString());

  /* Result:
     function(cx) {
       cx.fillStyle = 'red';
       cx.moveTo(0,0);
       cx.lineTo(1,10);
       cx.fill();
     }
  */
¶

fn.compile()
¶

Begins compilation.

 
  C3.fn.compile = function() {
    this.commands = [];
    $.extend(this, C3.Compiler);
    return this;
  };

  C3.Compiler = {
¶
  • Temporary getAttr() replacement: does nothing.
 
    getAttr: function(attr) {
      return undefined;
    },
¶
  • Temporary setAttr() replacement to cache the attribute set commands.
 
    setAttr: function(attr, value) {
      this.commands.push("cx." + attr + " = " + JSON.stringify(value) + ";");
      return this;
    },
¶
  • Temporary run() replacement to cache the method calls.
 
    run: function(method, args) {
      function toArgs(args) {
        var str = JSON.stringify(Array.prototype.slice.call(args, 0));
        return str.substr(1, str.length - 2);
      };

      this.commands.push("cx." + method + "(" + toArgs(args) + ");");
      return this;
    },
¶

fn.compileEnd()
¶

Terminate compilation by returning commands to defaults, and baking the commands into an anonymous function.

 
    compileEnd: function() {
      var commands = this.commands;

      delete(this.commands);
      delete(this.getAttr);
      delete(this.setAttr);
      delete(this.run);
      delete(this.compileEnd);

      return new Function("canvas", "var cx = canvas.getContext('2d');\n" + commands.join("\n"));
    }
  };

I am a helper, I am not part of the public API. The .style(), .clearStyle(), and .paint() functions delegate to me.

My responsibility is to keep track of styles set on the canvas, reset them if necessary, and paint (fill/stroke) them when needed.
